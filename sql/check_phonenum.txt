create or replace procedure check_phonenum
(p_phonenum in varchar2,
 o_exists out boolean
)
is
begin
    select case when count(*) > 0 then true
        else false
    end
    into o_exists
    from user_info
    where phonenum = p_phonenum;

exception
    when no_data_found then
        o_exists := false;
end;
/
